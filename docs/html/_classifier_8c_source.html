64void Classifier_OnSensor(classifier_t *clas, uint8_t outNum) {
65 if (outNum >= CLASSIFIER_NUM_SENSORS) return;
66
67 /*
68 * Buscar la caja mas antigua cuyo proximo sensor esperado sea outNum.
69 * La lista esta ordenada por orden de llegada, por lo que la primera
70 * coincidencia es fisicamente la que llego al sensor.
71 */
72 for (uint8_t i = 0; i < clas->count; i++) {
73
74 if (clas->list[i].stage != outNum) continue;
75
76 uint8_t type = clas->list[i].type;
77
78 if (type == clas->boxType[outNum]) {
79 /* El tipo coincide con esta salida: intentar disparar el actuador */
80 if (clas->onTrigger(outNum)) {
81 /* Expulsion exitosa: eliminar la caja de la lista */
82 removeAt(clas, i);
83 return;
84 }
85 /*
86 * Brazo ocupado: la caja sigue fisicamente por la cinta sin ser
87 * expulsada. Caemos al avance de stage para seguirla hacia la
88 * proxima salida.
89 */
90 }
91
92 /* Tipo incorrecto o brazo ocupado: avanzar al proximo sensor */
93 clas->list[i].stage++;
94 if (clas->list[i].stage >= CLASSIFIER_NUM_SENSORS) {
95 /* La caja paso el ultimo sensor sin ser clasificada: descartar */
96 removeAt(clas, i);
97 }
98 return;
99 }
100
101 /*
102 * Ninguna caja de la lista esperaba este sensor. Posibles causas:
103 * - Evento espurio del sensor.
104 * - CMD_NEW_BOX perdido o aun no procesado.
105 * Se ignora silenciosamente para no corromper el estado del clasificador.
106 */
107}
