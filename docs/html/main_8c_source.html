139static void InitTMR0(void) {

235static void OnCommCmd(uint8_t cmdId, uint8_t *payload, uint8_t nBytes) {
236 SimuCinta_OnCmd(&simu, cmdId, payload, nBytes);
237
238 switch (cmdId) {
239
240 case 0x50: /* CMD_START: inicio o reanudacion de la cinta */
241 /*
242 * Si el clasificador ya esta activo (cinta estaba corriendo),
243 * actualizar boxType por si el usuario cambio las asignaciones
244 * antes de reanudar. Si cintaLista == 0 (post-RESET), el primer
245 * OnNewBox() llamara a Classifier_Init() con los tipos correctos.
246 */
247 if (cintaLista) {
248 for (uint8_t i = 0; i < 3; i++) {
249 classifier.boxType[i] = simu.boxType[i];
250 }
251 }
252 break;
253
254 case 0x51: /* CMD_STOP: pausa (cajas fisicamente siguen en la cinta) */
255 /* No resetear: preservar el rastreo para reanudar correctamente */
256 break;
257
258 case 0x53: /* CMD_RESET: cinta vacia, borrar estado del clasificador */
259 cintaLista = 0;
260 Classifier_Reset(&classifier);
261 break;
262 }
263}

318static void On2ms(void) {
319 GPIOR0 &= ~_BV(GPIOR00);
320 if (--time100ms == 0) {
321 time100ms = 50;
322 PINB = (1 << PINB5); /* Toggle LED PB5 */
323 }

343int main(void) {
344 cli();
345
346 time100ms = 50;
347
348 /* Configurar pines */
349 DDRB = (1 << PORTB5); /* PB5: LED heartbeat (salida) */
350 DDRD |= (1 << PD2); /* PD2: TRIGGER HC-SR04 (salida) */
351 DDRD &= ~(1 << PD3); /* PD3: ECHO HC-SR04 (entrada) */
352
353 /* Tipos de salida por defecto hasta recibir CMD_START */
354 uint8_t defaultBoxType[3] = {0, 0, 0};
355
356 /* Tabla de actuadores indexada por numero de salida */
357 actuators[0] = &act0;
358 actuators[1] = &act1;
359 actuators[2] = &act2;
360
361 /* Inicializar perifericos y modulos */
362 InitTMR0();
363 InitTMR1();
364 Comm_Init(&comm, OnCommCmd);
365 HCSR04_Init(&sensor, hw_trigger_write, hw_echo_read, hw_get_us);
366 SimuCinta_Init(&simu, &comm, OnNewBox, OnIRSensor);
367 Actuator_Init(&act0, 0, 50, hw_Act_Set);
368 Actuator_Init(&act1, 1, 50, hw_Act_Set);
369 Actuator_Init(&act2, 2, 50, hw_Act_Set);
370 Classifier_Init(&classifier, defaultBoxType, OnActuatorTrigger);
371
372 sei();
373
374 /* ---- Super-loop ---- */
375 while (1) {
376
377 /* Tareas de 2 ms: ejecutar solo cuando el tick del Timer0 esta activo */
378 if (GPIOR0 & _BV(GPIOR00)) {
379 On2ms();
380 SimuCinta_Process(&simu);
381 Actuator_Process(&act0);
382 Actuator_Process(&act1);
383 Actuator_Process(&act2);
384 }
385
386 /* Tareas libres (no periodicidad estricta) */
387 Comm_Process(&comm);
388 HCSR04_Process(&sensor);
389
390 /* Procesar resultado del sensor ultrasonico */
391 if (sensor.state == HCSR_READY) {
392 /* sensor.distance contiene la distancia medida en cm */
393 sensor.state = HCSR_IDLE;
394 }
395
396 if (sensor.state == HCSR_ERROR) {
397 /* El sensor no respondio dentro del timeout */
398 sensor.state = HCSR_IDLE;
399 }
400 }
